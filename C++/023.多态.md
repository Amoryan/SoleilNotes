## 1 多态

意味着调用成员函数时，会根据调用函数的对象的类型来执行不同的函数。

```c++
class Animal {
public:
    virtual void eat() {//虚函数,动态链接(后期绑定),告诉派生类不要链接

    }
    
    //virtual void eat() = 0 //纯虚函数,啥也不干，派生类自己去干    
};

class Dog : public Animal {
public:
    void eat() {
        cout << "狗吃狗粮" << endl;
    }
};

class Cat : public Animal {
public:
    void eat() {
        cout << "猫吃猫粮" << endl;
    }
};


int main() {
     Animal *animal;
    Dog dog;
    animal = &dog;
    animal->eat();

    Cat cat;
    animal = &cat;
    animal->eat();

    return 0;
}
```
